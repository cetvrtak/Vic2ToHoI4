country_event = {
	id = neighbor_war.4
	title = neighbor_war.4.t
	desc = neighbor_war.4.d
	picture = GFX_report_event_german_troops
	
	is_triggered_only = yes
	
	option = {
		name = neighbor_war.4.a
		FROM = {
			country_event = { id = neighbor_war.5 }
		}
		effect_tooltip = {
			every_owned_state = {
				limit = { is_core_of = FROM }
				FROM = { transfer_state = PREV.id }
			}
		}
		ai_chance = {
			factor = 0
		}
	}
	option = {
		name = neighbor_war.4.b
		FROM = {
			country_event = { id = neighbor_war.6 }
		}
		effect_tooltip = {
			FROM = {
				create_wargoal = {
					target = ROOT
					type = take_core_state
					generator = core_states
				}
			}
		}
	}
}
